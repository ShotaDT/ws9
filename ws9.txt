#!/bin/bash
if [ -z "$1" ] ||[ -z "$2" ];
then
        echo "argument is invalid"
else
        lines=$(wc -l < $2)
        per=$((lines*$1/100))
        echo print $per % of the file $2
        sort -R $2| head -$per
fi
